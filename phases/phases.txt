3. Phase 1 — Foundation: Log Collection & Storage
Goal
Get logs from two sources (Sysmon + Zeek) flowing into Elasticsearch, normalized into ECS (Elastic Common Schema).

Why Only Two Log Sources First
Writing ECS normalization parsers for every log type (O365, Firewalls, Web Servers) at once is a 4+ week trap. Sysmon and Zeek both have mature, community-built ECS pipelines ready to use. Prove the pipeline works with these two, then add others one by one.

Step 1.3: Logstash Pipeline for Sysmon
Use the existing community Sysmon ECS pipeline from the elastic/integrations GitHub repository. This maps all Sysmon event IDs to ECS field names automatically.

Critical Rule for Correlation: Use @timestamp (the ingestion time when Logstash receives the event) for correlation. Do not use the endpoint's local clock (event.created). This prevents the "silent basket killer" where clock drift breaks your attack timelines.

Key ECS fields your correlation engine depends on:

text
host.name          → Hostname (Primary basket key)
user.name          → Username (Primary basket key)
process.pid        → Process ID
process.parent.pid → Parent process (for process-tree linking)
network.destination.ip → Outbound IP (for C2 detection)
@timestamp         → Ingestion time (USE THIS FOR CORRELATION)

def get_new_events(es_client, last_run_time):
    query = {
        "query": {
            "range": {
                "@timestamp": {           # USE INGESTION TIME (not event.created)
                    "gte": last_run_time.isoformat(),

Step 2.3: Sigma Rule Matcher (detection/sigma_matcher.py) — CORRECTED
Do not write a custom evaluator. Use the official pySigma library with the Elasticsearch backend.

python
import yaml
from sigma.rule import SigmaRule
from sigma.backends.elasticsearch import ElasticsearchLuceneBackend
from sigma.pipelines.elasticsearch import ecs_windows_pipeline
from elasticsearch import Elasticsearch

# Load rules once at startup
def load_sigma_rules(rules_dir):
    rules = []
    for f in Path(rules_dir).glob("*.yml"):
        with open(f) as fp:
            raw = yaml.safe_load(fp)
            # pySigma object for query generation
            sigma_rule = SigmaRule.from_yaml(f.read_text())
            rules.append({
                "raw": raw,
                "sigma_obj": sigma_rule,
                "rule_id": raw["id"],
                "level": raw.get("level", "medium"),
                "tags": raw.get("tags", [])
            })
    return rules

def match_event_against_rules(es_client, event, rules):
    backend = ElasticsearchLuceneBackend()
    for rule in rules:
        # Generate ES Lucene query string
        pipeline = ecs_windows_pipeline()
        query_string = rule["sigma_obj"].to_backend(backend, pipeline)
        
        # Run the query against the last 1 minute to see if this specific event matches
        search_body = {
            "query": {
                "query_string": {"query": query_string}
            }
        }
        # Simplified check: if the rule matches *any* event in the last minute, we treat it as a trigger.
        result = es_client.search(index="sysmon-*", body=search_body, size=1)
        if result["hits"]["total"]["value"] > 0:
            return rule
    return None

Step 2.4: Basket Manager (detection/basket_manager.py) — The Core Logic
This creates baskets, appends events, and ensures restart-proof persistence.

python
import redis
import json
import uuid
from datetime import datetime, timedelta

BASKET_TTL_MINUTES = 10

def get_or_create_basket(pg_conn, redis_client, host, user, source_ip):
    # 1. Check Redis first (fast)
    redis_key = f"basket:{host}:{user}"
    cached = redis_client.get(redis_key)
    if cached:
        return json.loads(cached)

    # 2. Check PostgreSQL (restart-proof)
    row = pg_conn.execute(
        "SELECT * FROM incident_baskets WHERE host_name=%s AND user_name=%s AND status='open'",
        (host, user)
    ).fetchone()
    if row:
        basket = dict(row)
        redis_client.setex(redis_key, BASKET_TTL_MINUTES * 60, json.dumps(basket))
        return basket

    # 3. Create new basket (write to BOTH PG and Redis)
    basket_id = str(uuid.uuid4())
    basket = {
        "basket_id": basket_id,
        "host_name": host,
        "user_name": user,
        "source_ip": source_ip,
        "status": "open",
        "confidence_score": 0,
        "matched_stages": [],
        "created_at": datetime.utcnow().isoformat()
    }
    pg_conn.execute(
        "INSERT INTO incident_baskets VALUES (%s,%s,%s,%s,'open',0,'[]',NOW(),NOW())",
        (basket_id, host, user, source_ip)
    )
    redis_client.setex(redis_key, BASKET_TTL_MINUTES * 60, json.dumps(basket))
    return basket

def append_event_to_basket(pg_conn, redis_client, basket, event, sigma_match):
    # Add stage to basket
    stage = {
        "event_type": sigma_match["rule_name"],
        "mitre_technique": sigma_match["mitre"],
        "ingestion_time": event["@timestamp"]
    }
    basket["matched_stages"].append(stage)

    # Write to PG (source of truth)
    pg_conn.execute(
        "UPDATE incident_baskets SET matched_stages=%s, updated_at=NOW() WHERE basket_id=%s",
        (json.dumps(basket["matched_stages"]), basket["basket_id"])
    )

    # Write event to basket_events table
    pg_conn.execute(
        "INSERT INTO basket_events (basket_id, event_type, raw_event, mitre_technique, ingestion_time) VALUES (%s,%s,%s,%s,%s)",
        (basket["basket_id"], sigma_match["rule_name"], json.dumps(event), str(sigma_match["mitre"]), event["@timestamp"])
    )

    # Update Redis
    redis_key = f"basket:{basket['host_name']}:{basket['user_name']}"
    redis_client.setex(redis_key, BASKET_TTL_MINUTES * 60, json.dumps(basket))

Step 2.6: Confidence Scoring & Tiered Alerting (detection/tiering.py)
python
def calculate_confidence(basket, chain):
    matched = len(basket["matched_stages"])
    total = len(chain["stages"])
    score = int((matched / total) * 100)
    basket["confidence_score"] = score

    if score == 100:
        return "critical"    # full chain matched
    elif score >= 75:
        return "high"
    elif score >= 50:
        return "medium"
    elif score >= 25:
        return "low"
    return None

def should_fire_alert(basket, sigma_match):
    # Tier 0: Critical single event — fire instantly, no basket needed
    if sigma_match["level"] == "critical":
        return "tier0", "critical"

    tier = calculate_confidence(basket, get_chain_for_basket(basket))
    if tier:
        return "tier_chain", tier
    return None, None
Step 2.7: Deduplication (Prevents Alert Storms)

def is_duplicate(redis_client, host, rule_id, window_seconds=300):
    dedup_key = f"dedup:{host}:{rule_id}"
    if redis_client.get(dedup_key):
        # Already fired recently — increment counter instead
        redis_client.incr(f"dedup_count:{host}:{rule_id}")
        return True
    redis_client.setex(dedup_key, window_seconds, 1)
    return False

# enrichment/cache.py
from datetime import datetime, timedelta

CACHE_TTL_HOURS = 24

def get_cached_result(pg_conn, indicator, source):
    row = pg_conn.execute(
        "SELECT verdict, checked_at FROM enrichment_cache WHERE indicator=%s AND source=%s",
        (indicator, source)
    ).fetchone()

    if row and (datetime.utcnow() - row["checked_at"]) < timedelta(hours=CACHE_TTL_HOURS):
        return row["verdict"]   # Return cached, skip API call
    return None                 # Cache miss, call the API

def set_cached_result(pg_conn, indicator, indicator_type, source, verdict):
    pg_conn.execute("""
        INSERT INTO enrichment_cache (indicator, indicator_type, source, verdict, checked_at)
        VALUES (%s, %s, %s, %s, NOW())
        ON CONFLICT (indicator, source) DO UPDATE SET verdict=%s, checked_at=NOW()
    """, (indicator, indicator_type, source, json.dumps(verdict), json.dumps(verdict)))

def extract_sha256_from_sysmon(event):
    hashes_str = event.get("Hashes", "")
    for part in hashes_str.split(","):
        if part.startswith("SHA256="):
            return part.replace("SHA256=", "")
    return None

# ai/rag_builder.py
import chromadb
import json
import requests

def build_mitre_knowledge_base():
    client = chromadb.PersistentClient(path="./chromadb_store")
    collection = client.get_or_create_collection("mitre_attack")

    # Download MITRE ATT&CK STIX data (free JSON from GitHub)
    mitre_data = requests.get(
        "https://raw.githubusercontent.com/mitre/cti/master/enterprise-attack/enterprise-attack.json"
    ).json()

    for obj in mitre_data["objects"]:
        if obj["type"] == "attack-pattern":    # These are the TTP technique objects
            technique_id = next(
                (ref["external_id"] for ref in obj.get("external_references", [])
                 if ref["source_name"] == "mitre-attack"), None
            )
            if technique_id:
                collection.add(
                    documents=[obj["description"]],
                    metadatas=[{"technique_id": technique_id, "name": obj["name"]}],
                    ids=[obj["id"]]
                )
    print(f"Loaded {collection.count()} MITRE techniques into ChromaDB")

Step 4.3: LangChain Narrative Generator (ai/narrator.py)
python
from langchain_community.llms import Ollama
from langchain_community.vectorstores import Chroma
from langchain.chains import RetrievalQA
from langchain_community.embeddings import OllamaEmbeddings

def generate_incident_narrative(basket, enrichment, matched_chain):
    # 1. Load the ChromaDB knowledge base
    embeddings = OllamaEmbeddings(model="phi3")
    vectorstore = Chroma(
        persist_directory="./chromadb_store",
        embedding_function=embeddings,
        collection_name="mitre_attack"
    )

    # 2. Build context string from the incident
    context = f"""
    Host: {basket['host_name']}
    User: {basket['user_name']}
    Attack chain matched: {matched_chain['name']}
    Confidence: {basket['confidence_score']}%
    Stages detected: {json.dumps(basket['matched_stages'], indent=2)}
    Enrichment results: {json.dumps(enrichment, indent=2)}
    """

    # 3. Retrieve relevant MITRE technique descriptions from ChromaDB
    techniques = [stage.get("mitre_technique") for stage in basket["matched_stages"]]
    retriever = vectorstore.as_retriever(search_kwargs={"k": len(techniques)})

    # 4. Call the local LLM via LangChain
    llm = Ollama(model="phi3")
    qa_chain = RetrievalQA.from_chain_type(llm=llm, retriever=retriever)

    prompt = f"""
    You are a SOC analyst assistant. Based on the following incident data, write:
    1. A 3-sentence plain-English summary of what the attacker did and how.
    2. The likely attack objective (ransomware / data theft / persistence / etc.).
    3. Immediate recommended containment actions (max 3 bullet points).

    Incident data:
    {context}

    Be concise. Do not repeat the raw data. Write for a junior SOC analyst.
    """

    return qa_chain.run(prompt)

from datetime import datetime

def update_baseline(pg_conn, user, source_ip, source_country, hour):
    """Called for every successful login event. Builds the normal pattern."""
    pg_conn.execute("""
        INSERT INTO login_baseline (user_name, source_ip, source_country,
            typical_hour_start, typical_hour_end, first_seen, last_seen, seen_count)
        VALUES (%s, %s, %s, %s, %s, NOW(), NOW(), 1)
        ON CONFLICT (user_name, source_ip) DO UPDATE SET
            last_seen = NOW(),
            seen_count = login_baseline.seen_count + 1,
            typical_hour_start = LEAST(login_baseline.typical_hour_start, %s),
            typical_hour_end = GREATEST(login_baseline.typical_hour_end, %s)
    """, (user, source_ip, source_country, hour, hour, hour, hour))

def check_anomaly(pg_conn, user, source_ip, source_country, login_time):
    hour = login_time.hour
    anomalies = []

    # Check 1: First-ever login from this IP for this user
    row = pg_conn.execute(
        "SELECT * FROM login_baseline WHERE user_name=%s AND source_ip=%s",
        (user, source_ip)
    ).fetchone()

    if not row:
        anomalies.append({
            "type": "first_seen_ip",
            "description": f"First login from {source_ip} for user {user}",
            "confidence": 30      # Low confidence alone
        })

    # Check 2: Login outside of normal hours
    if row and (hour < row["typical_hour_start"] or hour > row["typical_hour_end"]):
        anomalies.append({
            "type": "off_hours_login",
            "description": f"Login at {hour}:00 outside normal window ({row['typical_hour_start']}:00-{row['typical_hour_end']}:00)",
            "confidence": 40
        })

    # Check 3: Different country from baseline
    if row and source_country != row["source_country"]:
        anomalies.append({
            "type": "new_country",
            "description": f"Login from {source_country} — previously only seen from {row['source_country']}",
            "confidence": 60
        })

    return anomalies

import yaml
from pathlib import Path

def load_profile(client_name="default"):
    profile_path = Path(f"config/grc_profiles/{client_name}.yaml")
    if not profile_path.exists():
        profile_path = Path("config/grc_profiles/default.yaml")

    with open(profile_path) as f:
        return yaml.safe_load(f)

def filter_rules_by_profile(sigma_rules, profile):
    enabled_groups = profile.get("enabled_rule_groups", [])
    return [
        rule for rule in sigma_rules
        if any(group in rule.get("tags", []) for group in enabled_groups)
    ]

def is_auto_response_allowed(profile):
    return profile.get("auto_response_allowed", False)

# When analyst marks alert as FP in TheHive, this is called via webhook:
def suppress_alert(pg_conn, basket, analyst_username, suppress_days=7):
    from datetime import timedelta
    pg_conn.execute("""
        INSERT INTO alert_suppression (host_name, user_name, rule_id,
            suppressed_by, suppressed_at, expires_at)
        VALUES (%s, %s, %s, %s, NOW(), %s)
    """, (
        basket["host_name"],
        basket["user_name"],
        basket.get("primary_rule_id"),
        analyst_username,
        datetime.utcnow() + timedelta(days=suppress_days)
    ))

# Check suppression before firing any alert:
def is_suppressed(pg_conn, host, user, rule_id):
    row = pg_conn.execute("""
        SELECT 1 FROM alert_suppression
        WHERE host_name=%s AND user_name=%s AND rule_id=%s
        AND expires_at > NOW()
    """, (host, user, rule_id)).fetchone()
    return row is not None

import requests

THEHIVE_URL = "http://localhost:9000"
THEHIVE_API_KEY = "your_thehive_api_key"

def create_case(alert):
    case_payload = {
        "title": f"[{alert['tier'].upper()}] Attack Chain Detected on {alert['host']}",
        "description": alert["ai_narrative"],
        "severity": {"critical": 3, "high": 2, "medium": 1}.get(alert["tier"], 1),
        "tags": [alert["attack_chain"], f"confidence:{alert['confidence']}%"],
        "tasks": [
            {"title": "Review AI-generated attack timeline"},
            {"title": "Verify enrichment results (VT/AbuseIPDB)"},
            {"title": "Approve or reject auto-response action"}
        ]
    }
    requests.post(
        f"{THEHIVE_URL}/api/case",
        json=case_payload,
        headers={"Authorization": f"Bearer {THEHIVE_API_KEY}"}
    )

import subprocess
import os

def block_ip(ip, approved_by):
    # MVP approach 1: Print the command for manual execution (safest for demo)
    print(f"[ACTION REQUIRED] Run: sudo iptables -A OUTPUT -d {ip} -j DROP")
    
    # MVP approach 2: If running with sudo privileges, execute it.
    if os.geteuid() == 0:
        subprocess.run(["iptables", "-A", "OUTPUT", "-d", ip, "-j", "DROP"])
        subprocess.run(["iptables", "-A", "INPUT", "-s", ip, "-j", "DROP"])
        # Log the action for audit trail
        log_response_action(ip, "iptables_block", approved_by)
    else:
        # Log that we couldn't execute, but the alert still fired successfully.
